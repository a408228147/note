  public int coutline(String filepath){
         File file = new File(filepath);
         BufferedReader reader = null;
         int i= 0;
         try{
             reader = new BufferedReader(new FileReader(file));
             String line = null;
             boolean status = true;//判断是否处于注释当中 true不在注释中
             while ((line = reader.readLine()) != null) {
                line = line.trim();
                 if (line.startsWith("/")||line.length()==0||line.startsWith("*/"))
                 {
                     if (line.startsWith("/*"))
                     {
                         status = false;
                     }
                     if (line.startsWith("*/"))
                     {
                         status = true;
                     }
                 }else{
                     if (status) {
                         i++;
                     }
                 }
             }
         }catch(Exception e){
             e.printStackTrace();
         }
         return i;
     }
